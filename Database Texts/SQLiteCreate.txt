CREATE TABLE Media_Item (ID CHAR(10) NOT NULL, Copy_Number INT NOT NULL, Name VARCHAR(12), Year INT, Length VARCHAR(12), Audiobook_Flag INT, Album_Flag INT, Publisher VARCHAR(12), Movie_Flag INT, Director VARCHAR(12), Game_Flag INT, Console VARCHAR(12), Rating INT, Status VARCHAR(12), Order_Number INT NOT NULL, Due_Date DATE, Email_Address VARCHAR(15) NOT NULL);
CREATE TABLE Media_Genres (ID CHAR(10) NOT NULL, Genres VARCHAR(12) NOT NULL);
CREATE TABLE Artist_Genres (Name VARCHAR(12) NOT NULL, Genres VARCHAR(12) NOT NULL);
CREATE TABLE Branch (City VARCHAR(12) NOT NULL, State VARCHAR(12) NOT NULL);
CREATE TABLE Orders (Order_Number INT NOT NULL, Price VARCHAR(10), Quantity INT, Arrival_Date DATE, City VARCHAR(12) NOT NULL, State VARCHAR(12) NOT NULL);
CREATE TABLE Patron (Email_Address VARCHAR(15) NOT NULL, First_Name VARCHAR(12) NOT NULL, Last_Name VARCHAR(12) NOT NULL, Address VARCHAR(12), City VARCHAR(12) NOT NULL, State VARCHAR(12) NOT NULL);
CREATE TABLE Chapter (Name VARCHAR(12) NOT NULL, Number INT NOT NULL, ID CHAR(10) NOT NULL);
CREATE TABLE Author (Name VARCHAR(12) NOT NULL, Age INT);
CREATE TABLE Actor (Name VARCHAR(12) NOT NULL, Sex CHAR(1) NOT NULL, Age INT);
CREATE TABLE Game_Studio (Name VARCHAR(12) NOT NULL);
CREATE TABLE Artist (Name VARCHAR(12) NOT NULL);
CREATE TABLE Track (Name VARCHAR(12) NOT NULL, Genre VARCHAR(12), ArtistName VARCHAR(12) NOT NULL);
CREATE TABLE Author_Writes (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL);
CREATE TABLE Actor_Stars (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL);
CREATE TABLE Studio_Creates (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL);
CREATE TABLE Artist_Authors (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL);
CREATE TABLE Album_Contains (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL);
