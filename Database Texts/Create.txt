CREATE TABLE Media_Item ();
CREATE TABLE Media_Genres (ID CHAR(10) NOT NULL, Genres VARCHAR(12) NOT NULL, PrimaryKey(ID, Genres), ForeignKey(Genres) REFERENCES Media_Item(ID));
CREATE TABLE Artist_Genres (Name VARCHAR(12) NOT NULL, Genres VARCHAR(12) NOT NULL, PrimaryKey(Name, Genres), ForeignKey(Name) REFERENCES Artist(Name));
CREATE TABLE Branch (City VARCHAR(12) NOT NULL, State VARCHAR(12) NOT NULL, PrimaryKey(City, State));
CREATE TABLE Order ();
CREATE TABLE Patron ();
CREATE TABLE Chapter ();
CREATE TABLE Author (Name VARCHAR(12) NOT NULL, Age INT, PrimaryKey(Name));
CREATE TABLE Actor (Name VARCHAR(12) NOT NULL, Sex CHAR(1) NOT NULL, Age INT, PrimaryKey(Name));
CREATE TABLE Game_Studio (Name VARCHAR(12) NOT NULL, PrimaryKey(Name));
CREATE TABLE Artist (Name VARCHAR(12) NOT NULL, PrimaryKey(Name));
CREATE TABLE Track ();
CREATE TABLE Author_Writes (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL, PrimaryKey(Name, ID), ForeignKey(Name) REFERENCES Author(Name), ForeignKey(ID) REFERENCES Media_Item(ID));
CREATE TABLE Actor_Stars (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL, PrimaryKey(Name, ID), ForeignKey(Name) REFERENCES Actor(Name), ForeignKey(ID) REFERENCES Media_Item(ID));
CREATE TABLE Studio_Creates (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL, PrimaryKey(Name, ID), ForeignKey(Name) REFERENCES Game_Studio(Name), ForeignKey(ID) REFERENCES Media_Item(ID));
CREATE TABLE Artist_Authors (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL, PrimaryKey(Name, ID), ForeignKey(Name) REFERENCES Artist(Name), ForeignKey(ID) REFERENCES Media_Item(ID));
CREATE TABLE Album_Contains (Name VARCHAR(12) NOT NULL, ID CHAR(10) NOT NULL, PrimaryKey(Name, ID), ForeignKey(Name) REFERENCES Track(Name), ForeignKey(ID) REFERENCES Media_Item(ID));
